def ==(other)
  return false if self.size != other.size || @value != other.value

  @nodes.keys.each do |key|
    return false unless other.keys.include?(key)
  end

  @nodes.keys.each do |key|
    return false if @nodes[key] != other.nodes[key]
  end

  true
end
