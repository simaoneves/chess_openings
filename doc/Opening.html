def initialize(name, eco_code, moves)

  @moves = moves.map! { |move| move.is_a?(String)? move.to_sym : move }
end

def to_pgn
  result = ""
  index = 1
  @moves.each do |move|

    if index.odd?
      result += "#{(index / 2.0).ceil}.#{move}"
    elsif index.even? && @moves.size != index
      result += " #{move} "
    else
      result += " #{move}"
    end

    index += 1
  end
  return result
end
